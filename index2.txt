  <script>
    // 生成漂浮爱心（数量可调整）
    function createFloatingHearts() {
      const heartCount = 12; // 爱心数量
      for (let i = 0; i < heartCount; i++) {
        const heart = document.createElement('div');
        heart.className = 'floating-heart';
        heart.innerText = '♥';
        
        // 随机大小（12px-24px）
        heart.style.fontSize = (Math.random() * 12 + 12) + 'px';
        // 随机水平位置
        heart.style.left = (Math.random() * 100) + 'vw';
        // 随机动画时长（4-8秒）
        heart.style.animationDuration = (Math.random() * 4 + 4) + 's';
        // 随机延迟出现
        heart.style.animationDelay = (Math.random() * 8) + 's';
        
        document.body.appendChild(heart);
      }
    }

    // 页面加载时生成爱心
    createFloatingHearts();

    // 日期验证逻辑（修改为跳转独立页面）
    const targetDate = "2025.08.29"; 
    // 替换为 message.html 实际的线上访问地址（本地测试或部署后需更新）
    const messagePageUrl = "message.html"; 

    function checkDate() {
        const input = document.getElementById('dateInput').value;
        const tips = document.getElementById('tips');
        if (input === targetDate) {
            // 跳转到 message.html 独立页面
            window.location.href = messagePageUrl; 
        } else {
            tips.innerText = "日期不对哦～ 要输入正确日期才能开启";
        }
    }
  </script>
